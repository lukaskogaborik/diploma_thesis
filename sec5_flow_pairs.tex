\begin{lemma}
	Let $G$ be a cyclically $4$-edge connected graph with minimum degree at least $3$. Consider a pair of vertices $u, v\in V(G)$ and a graph $G'$, which arises from $G$ by adding the edge $uv$. If $G'$ has a Chebyshev NZ $2$-flow $\varphi$ satisfying
	\begin{enumerate}[label=(\alph*)]
		\item flow values are integral pairs -- $\im(\varphi)\subseteq\mathbb Z\times\mathbb Z$,
		\item $\varphi\left(\vv{uv}\right)=(0,1)$, and
		\item these are not the flow configurations around $u$ and $v$,
		\begin{figure}[h]
			\centering
			\input images/oddness_2_forbidden_config.tex
			\label{fig:odd_2_forbid}
		\end{figure}
	\end{enumerate}
	then $G$ has a $1/2$-flow-pair.
\end{lemma}

\begin{proof}
	% \textbf{TODO reference -- graph with flow number $4$ has a NZ $(\mathbb Z_2\times\mathbb Z_2)$-flow}
	
	% Take an NZ $(\mathbb Z_2\times\mathbb Z_2)$-flow $\varphi=(\varphi_1, \varphi_2)$ of $G'$. Since for any non-zero element $x$ of $\mathbb Z_2\times\mathbb Z_2$ there exists an automorphism mapping $x$ to $(0, 1)$, we can WLOG assume that $\varphi\left(\vv{uv}\right)=(0,1)$.
	
	% \textbf{TODO reference -- graph with NZ $\mathbb Z_2$-flow has a NZ $2$-flow with equivalent modulo}
	
	% By the equivalence between integral and group flows, $G'$ has also a two-dimensional Chebyshev NZ flow $\varphi'=(\varphi'_1, \varphi'_2)$ such that $\varphi'_i\big(\vv d\big)\equiv\varphi_i\big(\vv d\big)\pmod{2}$ for any dart $\vv d\in D(G')$ and $i\in\{1, 2\}$. Hence, $\varphi'\left(\vv{uv}\right)\in\{(0,1), (0,-1)\}$. By changing signs of all flow values in $G'$ we get the other option, and therefore WLOG $\varphi'\left(\vv{uv}\right)=(0,1)$.
	
	Multiply the second coordinate of $\varphi$ by two to obtain $\varphi'=(\varphi'_1,\varphi'_2)$. Observe that the flow values of $\varphi'$ are feasible for a $1/2$-flow-pair and $\varphi'\left(\vv{uv}\right)=(0,2)$. Since we are constructing a $1/2$-flow-pair on $G=G'-uv$, the rest of the proof consists of finding two $uv$-paths, where we can send the flow $(0,1)$ and get rid of the edge $e$.
	
	Let $H$ be an oriented graph such that $V(H)=V(G')$ and
		$$E(H)=\{\vv{xy}\mid xy\in E(G') \land (\varphi_1(\vv{xy})\neq 0 \lor \varphi_2(\vv{xy})=2)\}.$$
	One can easily check that this corresponds to set of orientations, in which we can adjust the flow $\varphi'$ by sending a value $(0,1)$ and the flow values remain feasible. However, some edges in $G'$ induce two darts in $H$. But if we take both of these and send $(0,1)$ through each of them, the resulting flow value on the edges is the same as before the process. Hence, if we find two dart-disjoint $uv$-paths in $H-\vv{uv}$, together with edge $uv$ they form two cycles. After sending $(0,1)$ along both of these cycles, the flow value at the edge $uv$ becomes zero and all other flow values remain feasible, and therefore we obtain a $1/2$-flow-pair of $G$.
	
	From the oriented version of Menger's Theorem \ref{th:oriented_menger}, there exist two dart-disjoint $\vv{uv}$-paths in $H-\vv{uv}$ if and only if each $uv$-dart-separator contains at least $2$ darts from the component containing $u$ to the component containing $v$. So it remains to prove this condition.
	
	Consider any $uv$-dart-separator $\gamma$. Denote the darts from the component containing $u$ to the component containing $v$ as $\gamma^+$.
	\begin{itemize}
		\item If there is an edge having flow value with non-zero first coordinate in $\gamma$, there must be at least one more due to flow conservation constraint. Both of them are in $H$ in both orientations.
		\item If $|\gamma|\geq 5$, the number of incoming $(0,2)$ edges must be equal to the number of outgoing $(0,2)$ edges. Therefore $|\gamma|\geq 6$, $|\gamma^+|\geq 3$, so there are at least two darts different from $\vv{uv}$. 
		\item If $|\gamma|\leq 4$, the size of cut $\gamma'$ in $H-\vv{uv}$ is equal to $3$. Since the underlying graph $G$ is cyclically $4$-edge connected, at least one of the components must be acyclic mutlipole. Acyclic multipole with vertices of degrees $d_1, \dots, d_k$ has $\sum_{i=1}^k d_i - k$ hanging edges. Together with the fact that $d_i\geq 3$, it has at least $2k$ hanging edges. With the fact that $|\gamma'|=3$, we get $k=1$. This leads to two forbidden configurations -- cut around $u$ and cut around $v$.
	\end{itemize}
\end{proof}

\begin{corollary}
	Each snark of oddness $2$ has a $1/2$-flow-pair.
\end{corollary}
